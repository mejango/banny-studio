### Task 2: `Settings.autoFrame` (BannyCore) + tests
Add to `Sources/BannyCore/Document.swift` (Settings extension) or new file `Sources/BannyCore/AutoFrame.swift`:

```swift
public extension Settings {
    /// Frame (w, h) to adopt when a backdrop is added, or nil to leave alone.
    /// Fires only for the first backdrop while the frame is the untouched
    /// 16:9 default. Returns the asset's reduced pixel ratio.
    func autoFrame(assetPixelW: Int, assetPixelH: Int, hasBackgroundCues: Bool) -> (w: Double, h: Double)? {
        guard !hasBackgroundCues, frameW == 16, frameH == 9,
              assetPixelW > 0, assetPixelH > 0 else { return nil }
        func gcd(_ a: Int, _ b: Int) -> Int { b == 0 ? a : gcd(b, a % b) }
        let g = gcd(assetPixelW, assetPixelH)
        return (Double(assetPixelW / g), Double(assetPixelH / g))
    }
}
```

Tests (`Tests/BannyCoreTests/AutoFrameTests.swift`): 1200×1200 → (1,1); 1200×1390 → (120,139); nil when hasBackgroundCues; nil when frame isn't 16×9; nil for zero dims.

```swift
func addBundledBackdrop(url: URL) {
    // Reuse an already-imported copy (match by source filename) before adding bytes.
    let sourceName = url.lastPathComponent
    if let existing = document.assets.first(where: { $0.name == BuiltInBackdrops.displayName(sourceName) }) {
        addBackgroundCue(assetID: existing.id, assetName: existing.name)
        applyAutoFrameIfNeeded(assetID: existing.id)   // no-op: cues now exist
        return
    }
    guard let data = try? Data(contentsOf: url) else { return }
    let hadCues = !scene.backgroundTracks.flatMap(\.cues).isEmpty
    guard let asset = addAsset(data: data, ext: "gif", name: BuiltInBackdrops.displayName(sourceName)) else { return }
    addBackgroundCue(assetID: asset.id, assetName: asset.name)
    applyAutoFrame(assetID: asset.id, hadCuesBeforeAdd: hadCues)
}
```
